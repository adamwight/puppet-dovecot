function dovecot::print_config_value($value) {
  $quotedValue = $value ? {
    Undef   => '',
    ''      => '""',
    true    => 'yes',
    false   => 'no',
    /(#|\n|\A\s+|\s+\z)/ => "\"${value}\"",
    default => $value,
  }
}
